def plot_sales(SalesTotalVentaDate, x, y, title="", xlabel='Fecha', ylabel='Ventas', dpi=80):
    plt.figure(figsize=(16,5), dpi=dpi)
    plt.plot(x, y, color='tab:red')
    plt.gca().set(title=title, xlabel=xlabel, ylabel=ylabel)
    plt.show()
